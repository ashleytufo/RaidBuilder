CREATE TABLE "rb_players" (
	"ID"	INTEGER NOT NULL,
	"Discord_Name"	TEXT,
	"Abrv"	TEXT,
	"Aliases"	TEXT,
	"Toons"	TEXT,
	"RaidLead"	INTEGER,
	"Guildie"	INTEGER,
	"Buyer"	INTEGER,
	"Carry"	INTEGER,
	"Favorite"	INTEGER,
	"Shitlist"	INTEGER,
	"Priority"	INTEGER,
	PRIMARY KEY("ID" AUTOINCREMENT)
)

CREATE TABLE "rb_roles" (
	"ID"	INTEGER,
	"IconName"	TEXT,
	"Name"	TEXT,
	"Class"	TEXT,
	"Tag"	TEXT,
	"IconPNG"	TEXT,
	PRIMARY KEY("ID" AUTOINCREMENT)
)

CREATE TABLE "rb_rosters" (
	"ID"	INTEGER,
	"Raid"	TEXT,
	"Group1"	BLOB,
	"Group2"	BLOB,
	"Group3"	BLOB,
	"Group4"	BLOB,
	"Group5"	BLOB,
	"Group6"	BLOB,
	"Group7"	BLOB,
	"Group8"	BLOB,
	"Standby1"	BLOB,
	"Standby2"	BLOB,
	PRIMARY KEY("ID" AUTOINCREMENT)
)

CREATE TABLE "rb_signees" (
	"ID"	INTEGER NOT NULL,
	"Signup_ID"	INTEGER NOT NULL,
	"Discord_Name"	TEXT NOT NULL,
	"Class"	TEXT,
	"Role"	TEXT,
	"Status"	TEXT,
	PRIMARY KEY("ID" AUTOINCREMENT)
)

CREATE TABLE "rb_signups" (
	"ID"	INTEGER NOT NULL,
	"Signup_Name"	TEXT NOT NULL UNIQUE,
	"Name_Shortcode"	TEXT NOT NULL UNIQUE,
	"Raid_Date_Time"	TEXT NOT NULL,
	"Signup_Text"	TEXT NOT NULL,
	PRIMARY KEY("ID" AUTOINCREMENT)
)
